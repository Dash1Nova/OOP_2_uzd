773// An assertion predicate determining whether a one string is a prefix for
774// another.
775template <typename StringType>
776AssertionResult HasPrefix(const StringType& str, const StringType& prefix) {
777 if (str.find(prefix, 0) == 0) return AssertionSuccess();
778
779 const bool is_wide_string = sizeof(prefix[0]) > 1;
780 const char* const begin_string_quote = is_wide_string ? "L\"" : "\"";
781 return AssertionFailure()
782 << begin_string_quote << prefix << "\" is not a prefix of "
783 << begin_string_quote << str << "\"\n";
784}
