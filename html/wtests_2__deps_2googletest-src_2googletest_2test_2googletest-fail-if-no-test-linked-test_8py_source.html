47 def Run(self, program_name, flag=None, env=None):
48 """Run the given program with the given flag.
49
50 Args:
51 program_name: Name of the program to run.
52 flag: The command line flag to pass to the program, or None.
53 env: Dictionary with environment to pass to the subprocess.
54
55 Returns:
56 True if the program exits with code 0, false otherwise.
57 """
58
59 exe_path = gtest_test_utils.GetTestExecutablePath(program_name)
60 args = [exe_path]
61 if flag is not None:
62 args += [flag]
63 process = gtest_test_utils.Subprocess(args, capture_stderr=False, env=env)
64 return process.exited and process.exit_code == 0
